<div class="container">

    <h2>🎧 Mi Sonido Pro</h2>

    

    <label>Evento:</label>

    <input type="text" id="nombreEvento" placeholder="Ej: Boda Familia Sánchez">

    <input type="date" id="fechaEvento">

    <input type="text" id="lugarEvento" placeholder="Ciudad / Municipio">


    <hr style="border: 0; border-top: 1px solid #333; margin: 30px 0;">


    <h3>🚚 Costos y Ganancias</h3>

    <label>Distancia total (Km ida y vuelta):</label>

    <input type="number" id="distancia" placeholder="Ej: 100" oninput="calcularTotal()">

    

    <div style="display: flex; gap: 10px;">

        <div style="flex: 1;"><label>Rendimiento (Km/L):</label><input type="number" id="rend" placeholder="Ej: 8" oninput="calcularTotal()"></div>

        <div style="flex: 1;"><label>Precio Litro ($):</label><input type="number" id="precio" placeholder="Ej: 24" oninput="calcularTotal()"></div>

    </div>


    <label>Casetas ($):</label>

    <input type="number" id="casetas" placeholder="Total peajes" oninput="calcularTotal()">


    <label>Staff (Ayudantes x Pago):</label>

    <div style="display: flex; gap: 10px;">

        <input type="number" id="numStaff" placeholder="Cantidad" oninput="calcularTotal()">

        <input type="number" id="pagoStaff" placeholder="$ por cada uno" oninput="calcularTotal()">

    </div>


    <hr style="border: 0; border-top: 1px dashed var(--color-principal); margin: 20px 0;">

    

    <label style="color: var(--color-principal); font-weight: bold; font-size: 16px;">¿Cuánto cobrarás en total al cliente? ($):</label>

    <input type="number" id="cobroTotal" placeholder="Ej: 8000" oninput="calcularTotal()" style="border: 2px solid var(--color-principal);">


    <div id="previoTotal" style="text-align: center; margin: 25px 0; font-size: 1.2em; font-family: sans-serif;">

        <div style="color: #ccc;">Gastos Operativos: $0.00</div>

        <div style="color: var(--color-principal); font-size: 1.5em; font-weight: bold; margin-top: 10px;">Ganancia Libre: $0.00</div>

    </div>


    <button onclick="guardarEvento()">💾 Guardar y Agendar</button>


    <hr style="border: 0; border-top: 1px solid #333; margin: 30px 0;">

    <div id="listaEventos"></div>

<script>

    document.addEventListener('DOMContentLoaded', mostrarEventos);


    function calcularTotal() {

        const d = parseFloat(document.getElementById('distancia').value) || 0;

        const r = parseFloat(document.getElementById('rend').value) || 0;

        const p = parseFloat(document.getElementById('precio').value) || 0;

        const c = parseFloat(document.getElementById('casetas').value) || 0;

        const nS = parseInt(document.getElementById('numStaff').value) || 0;

        const pS = parseFloat(document.getElementById('pagoStaff').value) || 0;

        const cobro = parseFloat(document.getElementById('cobroTotal').value) || 0;


        const gas = r > 0 ? (d / r) * p : 0;

        const staff = nS * pS;

        const totalGastos = gas + c + staff;

        const gananciaLimpia = cobro - totalGastos;


        document.getElementById('previoTotal').innerHTML = `

            <div style="color: #ccc;">Gastos Operativos: $${totalGastos.toFixed(2)}</div>

            <div style="color: var(--color-principal); font-size: 1.5em; font-weight: bold; margin-top: 10px;">Ganancia Libre: $${gananciaLimpia.toFixed(2)}</div>

        `;

        

        return { gasto: totalGastos.toFixed(2), ganancia: gananciaLimpia.toFixed(2), cobro: cobro.toFixed(2) };

    }


    function guardarEvento() {

        const nombre = document.getElementById('nombreEvento').value;

        const fecha = document.getElementById('fechaEvento').value;

        const lugar = document.getElementById('lugarEvento').value;

        const finanzas = calcularTotal();


        if(!nombre || !fecha) return alert("Por favor pon al menos el nombre y la fecha.");


        const nuevo = { id: Date.now(), nombre, fecha, lugar, gasto: finanzas.gasto, ganancia: finanzas.ganancia, cobro: finanzas.cobro };

        let eventos = JSON.parse(localStorage.getItem('sonidoApp')) || [];

        eventos.push(nuevo);

        localStorage.setItem('sonidoApp', JSON.stringify(eventos));


        document.querySelectorAll('input').forEach(i => i.value = '');

        calcularTotal();

        mostrarEventos();

    }


    function mostrarEventos() {

        const div = document.getElementById('listaEventos');

        div.innerHTML = '';

        let eventos = JSON.parse(localStorage.getItem('sonidoApp')) || [];

        eventos.sort((a,b) => new Date(a.fecha) - new Date(b.fecha));


        eventos.forEach(ev => {

            const msj = `Hola! Te comparto la cotización para tu evento:%0A*${ev.nombre}*%0A📍 Lugar: ${ev.lugar}%0A📅 Fecha: ${ev.fecha}%0A💰 Total a pagar: $${ev.cobro}%0A¡Quedo a tus órdenes!`;

            

            div.innerHTML += `

                <div class="event-card">

                    <strong style="font-size: 1.2em; color: var(--color-principal);">${ev.nombre}</strong><br>

                    <div style="margin: 10px 0; color: #ccc;">📅 ${ev.fecha} | 📍 ${ev.lugar}</div>

                    <div style="background: #111; padding: 8px; border-radius: 5px;">

                        Cobro Cliente: $${ev.cobro}<br>

                        Gastos: -$${ev.gasto}<br>

                        <strong style="color: var(--color-principal);">Ganancia: $${ev.ganancia}</strong>

                    </div>

                    <button class="wa-btn" onclick="window.open('https://wa.me/?text=${msj}', '_blank')">📱 Enviar Cotización al Cliente</button>

                    <button class="btn-delete" onclick="borrar(${ev.id})">×</button>

                </div>

            `;

        });

    }


    function borrar(id) {

        if(confirm("¿Seguro que quieres borrar este evento?")) {

            let eventos = JSON.parse(localStorage.getItem('sonidoApp'));

            eventos = eventos.filter(e => e.id !== id);

            localStorage.setItem('sonidoApp', JSON.stringify(eventos));

            mostrarEventos();

        }

    }
